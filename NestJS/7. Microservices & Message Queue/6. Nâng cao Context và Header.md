Trong các hệ thống phức tạp (như Kafka, NATS), bạn cần lấy thông tin Header hoặc Topic. Sử dụng `@Payload()` và `@Ctx()`.

```typescript
import { MessagePattern, Payload, Ctx, KafkaContext } from '@nestjs/microservices';

@Controller()
export class AnalyticsController {
  @MessagePattern('user.activity')
  getAnalytics(@Payload() data: any, @Ctx() context: KafkaContext) {
    const originalTopic = context.getTopic();
    const headers = context.getMessage().headers;
    console.log(`Received from topic: ${originalTopic}`);
    return { processed: true };
  }
}
